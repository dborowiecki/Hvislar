CREATE TABLE account (
	id_account_pk SERIAL UNIQUE,
	username VARCHAR(40) NOT NULL,
	passwd VARCHAR(256) NOT NULL,
	email VARCHAR(50) NOT NULL UNIQUE,
	dos DATE not null default CURRENT_DATE,
	last_time_logged TIMESTAMP default CURRENT_DATE
);

CREATE TABLE conversation (
	id_conversation_pk SERIAL UNIQUE
);

CREATE TABLE contact (
	id_contact_pk int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
	id_user_fk INTEGER REFERENCES account(id_account_pk) NOT NULL,
	id_account_fk INTEGER REFERENCES account(id_account_pk) NOT NULL,
	contact_name VARCHAR(20) DEFAULT 'null',
	status BOOL default true
);

CREATE TABLE contact_list (
	id_account_fk INTEGER REFERENCES account(id_account_pk),
	id_contact_fk INTEGER REFERENCES contact(id_contact_pk)
);

CREATE TABLE message (
	id_message_pk SERIAL UNIQUE,
	id_conversation_fk INTEGER REFERENCES conversation(id_conversation_pk),
	content_of_msg VARCHAR(250) NOT NULL,
	send_time TIMESTAMP
);

CREATE TABLE messages_in_conversation (
	id_contact_fk INTEGER REFERENCES contact(id_contact_pk),
	id_message_fk INTEGER REFERENCES message(id_message_pk)
);
